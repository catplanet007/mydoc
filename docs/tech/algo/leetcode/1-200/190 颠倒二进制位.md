class Solution {
public:
    uint32_t reverseBits(uint32_t n) {
        uint32_t r = 0;
        for (int i = 0; i < 16; i++) {
            uint32_t mask = 1 << i;
            r |= (mask & n) << (32 - (i + 1) - i);
            uint32_t mask2 = 0x80000000 >> i;
            r |= (mask2 & n) >> (32 - (i + 1) - i);
        }
        return r;
    }
};

```cpp
class Solution {
public:
    vector<vector<int>> generate(int numRows) {
        vector<vector<int>> r;
        r.push_back({1});
        for (int i = 1; i < numRows; i++) {
            vector<int> vec(i + 1, 1);
            vector<int>& lastVec = r[i - 1];
            for (int j = 1; j < i; j++) {
                vec[j] = lastVec[j - 1] + lastVec[j];
            }
            r.push_back(vec);
        }
        return r;
    }
};
```

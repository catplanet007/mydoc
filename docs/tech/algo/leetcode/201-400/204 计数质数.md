```cpp
class Solution {
public:
    int countPrimes(int n) {
        if (n <= 1) {
            return 0;
        }
        vector<int> vec(n, 1);
        int res = 0;
        for (int i = 2; i < n; i++) {
            if (vec[i] == 0) {
                continue;
            }
            for (long j = long(i) * long(i); j < n; j += i) {
                vec[j] = 0;
            }
            res++;
        }
        return res;
    }
};
```

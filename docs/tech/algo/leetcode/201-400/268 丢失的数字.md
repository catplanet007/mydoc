cost: 2m20s

```cpp
class Solution {
public:
    int missingNumber(vector<int>& nums) {
        int r = 0;
        for (int i = 0; i < nums.size(); i++) {
            r += i;
            r -= nums[i];
        }
        r += nums.size();
        return r;
    }
};
```
